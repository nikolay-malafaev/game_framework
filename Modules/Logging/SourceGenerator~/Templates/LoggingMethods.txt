private void LogInfo(string message, UnityEngine.Object? context = null, params string[] tags)
{
    context ??= (object)this as UnityEngine.Object;
    GameFramework.Logging.LoggingContainer.Context.LoggingHelper.SendLog(UnityEngine.LogType.Log, LoggingTag, tags, message, context);
}
